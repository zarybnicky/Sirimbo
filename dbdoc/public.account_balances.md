```sql
CREATE MATERIALIZED VIEW account_balances AS (
 SELECT account.id,
    (COALESCE(account.opening_balance, 0.0) + COALESCE(sum(posting.amount), 0.0)) AS balance
   FROM (account
     LEFT JOIN posting ON ((account.id = posting.account_id)))
  GROUP BY account.id
)
```
